# v8_clean_backtest.py # pip install pandas requests import requests import pandas as pd from datetime import datetime SYMBOL = "BTCUSDT" INTERVAL = "15m" LIMIT = 1500 INITIAL_BALANCE = 300.0 FEE_RATE = 0.0005 MA_DENSE_THRESHOLD = 0.003 VOLUME_MULTIPLIER = 1.3 BREAKOUT_LOOKBACK = 20 ATR_PERIOD = 14 STOP_ATR_MULT = 1.0 TP1_R = 2.0 TP2_R = 4.0 TRAIL_ATR_MULT = 1.2 RISK_PCT = 0.01 def fetch_klines(symbol, interval, limit=1500): url = "https://api.binance.com/api/v3/klines" params = {"symbol": symbol, "interval": interval, "limit": limit} data = requests.get(url, params=params, timeout=10).json() df = pd.DataFrame(data, columns=[ "open_time", "open", "high", "low", "close", "volume", "close_time", "quote_volume", "trades", "taker_buy_base", "taker_buy_quote", "ignore" ]) for col in ["open", "high", "low", "close", "volume"]: df[col] = df[col].astype(float) df["time"] = pd.to_datetime(df["open_time"], unit="ms") return df def add_indicators(df): for n in [20, 60, 120]: df[f"ma{n}"] = df["close"].rolling(n).mean() df[f"ema{n}"] = df["close"].ewm(span=n, adjust=False).mean() df["volume_ma20"] = df["volume"].rolling(20).mean() high_low = df["high"] - df["low"] high_close = (df["high"] - df["close"].shift()).abs() low_close = (df["low"] - df["close"].shift()).abs() df["tr"] = pd.concat([high_low, high_close, low_close], axis=1).max(axis=1) df["atr"] = df["tr"].rolling(ATR_PERIOD).mean() return df def check_signal(df, i): row = df.iloc[i] ma_cols = ["ma20", "ma60", "ma120", "ema20", "ema60", "ema120"] if row[ma_cols].isna().any() or pd.isna(row["atr"]): return None ma_max = row[ma_cols].max() ma_min = row[ma_cols].min() ma_spread = (ma_max - ma_min) / row["close"] dense = ma_spread < MA_DENSE_THRESHOLD if not dense: return None volume_ok = row["volume"] > row["volume_ma20"] * VOLUME_MULTIPLIER if not volume_ok: return None recent_high = df["high"].iloc[i - BREAKOUT_LOOKBACK:i].max() recent_low = df["low"].iloc[i - BREAKOUT_LOOKBACK:i].min() if row["close"] > ma_max and row["close"] > recent_high: return "LONG" if row["close"] < ma_min and row["close"] < recent_low: return "SHORT" return None def calc_qty(balance, entry, stop): max_loss = balance * RISK_PCT risk_per_unit = abs(entry - stop) if risk_per_unit <= 0: return 0 return max_loss / risk_per_unit def backtest(df): balance = INITIAL_BALANCE position = None trades = [] peak_balance = balance max_drawdown = 0 start_i = max(130, BREAKOUT_LOOKBACK + 1) for i in range(start_i, len(df)): row = df.iloc[i] if position: side = position["side"] entry = position["entry"] qty = position["qty"] stop = position["stop"] tp1 = position["tp1"] tp2 = position["tp2"] exit_price = None exit_reason = None if side == "LONG": position["highest"] = max(position["highest"], row["high"]) trail_stop = position["highest"] - row["atr"] * TRAIL_ATR_MULT stop = max(stop, trail_stop) if row["low"] <= stop: exit_price = stop exit_reason = "STOP/TRAIL" elif row["high"] >= tp2: exit_price = tp2 exit_reason = "TP2" elif row["high"] >= tp1 and not position["tp1_hit"]: position["tp1_hit"] = True position["stop"] = entry continue if side == "SHORT": position["lowest"] = min(position["lowest"], row["low"]) trail_stop = position["lowest"] + row["atr"] * TRAIL_ATR_MULT stop = min(stop, trail_stop) if row["high"] >= stop: exit_price = stop exit_reason = "STOP/TRAIL" elif row["low"] <= tp2: exit_price = tp2 exit_reason = "TP2" elif row["low"] <= tp1 and not position["tp1_hit"]: position["tp1_hit"] = True position["stop"] = entry continue position["stop"] = stop if exit_price: if side == "LONG": pnl = (exit_price - entry) * qty else: pnl = (entry - exit_price) * qty fee = (entry * qty + exit_price * qty) * FEE_RATE pnl -= fee balance += pnl trades.append({ "time": row["time"], "side": side, "entry": entry, "exit": exit_price, "pnl": pnl, "balance": balance, "reason": exit_reason }) peak_balance = max(peak_balance, balance) dd = (peak_balance - balance) / peak_balance max_drawdown = max(max_drawdown, dd) position = None else: signal = check_signal(df, i) if not signal: continue entry = row["close"] atr = row["atr"] if signal == "LONG": stop = entry - atr * STOP_ATR_MULT risk = entry - stop tp1 = entry + risk * TP1_R tp2 = entry + risk * TP2_R qty = calc_qty(balance, entry, stop) position = { "side": "LONG", "entry": entry, "qty": qty, "stop": stop, "tp1": tp1, "tp2": tp2, "tp1_hit": False, "highest": row["high"] } elif signal == "SHORT": stop = entry + atr * STOP_ATR_MULT risk = stop - entry tp1 = entry - risk * TP1_R tp2 = entry - risk * TP2_R qty = calc_qty(balance, entry, stop) position = { "side": "SHORT", "entry": entry, "qty": qty, "stop": stop, "tp1": tp1, "tp2": tp2, "tp1_hit": False, "lowest": row["low"] } return balance, trades, max_drawdown def report(balance, trades, max_drawdown): wins = [t for t in trades if t["pnl"] > 0] losses = [t for t in trades if t["pnl"] <= 0] total_profit = sum(t["pnl"] for t in wins) total_loss = abs(sum(t["pnl"] for t in losses)) win_rate = len(wins) / len(trades) * 100 if trades else 0 pf = total_profit / total_loss if total_loss > 0 else 0 print("\n【V8 六均线密集突破回测报告】") print(f"交易对：{SYMBOL}") print(f"周期：{INTERVAL}") print(f"初始资金：{INITIAL_BALANCE:.2f}") print(f"最终资金：{balance:.2f}") print(f"收益率：{(balance / INITIAL_BALANCE - 1) * 100:.2f}%") print(f"最大回撤：{max_drawdown * 100:.2f}%") print(f"交易次数：{len(trades)}") print(f"胜率：{win_rate:.2f}%") print(f"Profit Factor：{pf:.2f}") print("\n最近10笔交易：") for t in trades[-10:]: print( f"{t['time']} | {t['side']} | " f"entry={t['entry']:.2f} exit={t['exit']:.2f} " f"pnl={t['pnl']:.4f} | {t['reason']}" ) if __name__ == "__main__": df = fetch_klines(SYMBOL, INTERVAL, LIMIT) df = add_indicators(df) balance, trades, max_drawdown = backtest(df) report(balance, trades, max_drawdown)
