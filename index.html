python v8_2_six_ma_breakout_backtest.py import requests import pandas as pd from dataclasses import dataclass from typing import Optional, List, Dict # ========================= # 参数区 # ========================= SYMBOL = "BTCUSDT" INTERVAL = "15m" LIMIT = 1500 INITIAL_BALANCE = 300.0 FEE_RATE = 0.0005 # 六均线密集 MA_DENSE_THRESHOLD = 0.008 # 六条均线最大距离 / close，小于0.8%算密集 DENSE_MIN_BARS = 10 # 连续密集至少10根K线 RANGE_LOOKBACK = 20 # 用最近20根K线作为密集区高低点 # 突破过滤 BREAKOUT_ATR_MULT = 0.3 # 收盘突破区间 + ATR*0.3 才确认 VOLUME_MULTIPLIER = 1.5 # 单笔完整计划最大亏损账户1% ENTRY_POSITION_PCT = 0.3 # 突破先开30% # 防双杀 COOLDOWN_BARS = 10 # 止损后冷却10根K线 # ========================= # 数据结构 # ========================= @dataclass class Position: side: str entry: float qty: float stop: float full_risk_qty: float range_high: float range_low: float breakout_level: float added: bool breakout_i: int highest: float = 0.0 lowest: float = 0.0 # ========================= # 拉币安K线 # ========================= def fetch_klines(symbol: str, interval: str, limit: int = 1500) -> pd.DataFrame: url = "https://api.binance.com/api/v3/klines" params = {"symbol": symbol, "interval": interval, "limit": limit} resp = requests.get(url, params=params, timeout=15) resp.raise_for_status() data = resp.json() df = pd.DataFrame(data, columns=[ "open_time", "open", "high", "low", "close", "volume", "close_time", "quote_volume", "trades", "taker_buy_base", "taker_buy_quote", "ignore" ]) for col in ["open", "high", "low", "close", "volume"]: df[col] = df[col].astype(float) df["time"] = pd.to_datetime(df["open_time"], unit="ms") return df #
